# 常见区间合并做法（三种）
## 第一种（最常见 初学者使用）  
``` c++
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef pair<ll,ll> PII;
vector<PII>segs;
vector<PII> merge(vector<PII>& segs)
{
    vector<PII>res;
    ll l=-2e9,r=-2e9;
    for(auto seg:segs)
    {
        //如果目前区间的右端点小于新区间的左端点
        //那么将当前区间上传并更新区间
        if(r<seg.first)
        {
            if(l!=-2e9)res.push_back({l,r});
            l=seg.first;
            r=seg.second;
        }
        //其他情况就是有交集，那么更新右端点,因为左端点已经在主函数里面排序排好了
        else{
            r=max(r,seg.second);
        }
    }
    //上传最后区间
    if(l!=-2e9) res.push_back({l,r});
    segs=res;
    return segs;
}
int main()
{
    int n;
    cin>>n;
    while(n--)
    {
        ll l,r;
        cin>>l>>r;
        segs.push_back({l,r});
    }
    sort(segs.begin(),segs.end());
    merge(segs);
    cout<<segs.size();
    return 0;
}
```
- 这种做法相较于第二种的缺点是定义了两个负无穷，导致加了两个if判断
- 第二种则直接将第一个区间作为初始，更加简洁且易懂
***  
## 第二种（比较巧妙 竞赛适用）
```c++
/*
给定 n个区间 [li,ri]，要求合并所有有交集的区间。
注意如果在端点处相交，也算有交集。
输出合并完成后的区间个数。
例如：[1,3]和 [2,6]可以合并为一个区间 [1,6]
输入格式
第一行包含整数 n,接下来 n行，每行包含两个整数 l和 r
输出格式
共一行，包含一个整数，表示合并区间完成后的区间个数。
数据范围
1≤n≤100000
−109≤li≤ri≤109*/
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef pair<ll,ll>PII;
vector<PII>segs;
vector<PII> merge(vector<PII> &segs)
{
  vector<PII>res;
  ll l=segs[0].first;
  ll r=segs[0].second;
  for(int i=1;i<segs.size();i++)
    {
      auto item=segs[i];
      if(r<item.first) 
        {
          res.push_back({l,r});
          l=item.first;
          r=item.second;
        }
      else 
        r=max(r,item.second);
    }
  res.push_back({l,r});
  segs=res;
  return segs;
}
int main()
{
  int n;
  cin>>n;
  while(n--)
    {
      ll l,r;
      cin>>l>>r;
      segs.push_back({l,r});
    }
  sort(segs.begin(),segs.end());
  merge(segs);
  cout<<segs.size();
  return 0;
}
```
