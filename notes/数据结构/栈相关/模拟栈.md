# 数据结构之模拟栈  
```c++
#include <bits/stdc++.h>
using namespace std;
const int N=1e5+10;
typedef long long ll;
ll stk[N],tt=-1;
void push(int x)
{
    stk[++tt]=x;
}
void pop()
{
    tt--;
}
bool empty()
{
    return tt==-1;
}
ll query()
{
    return stk[tt];
}
int main()
{
    int m;
    cin>>m;
    while(m--)
    {
        int x;
        string a;
        cin>>a;
        if(a=="push")
        {
            cin>>x;
            push(x);
        }
        if(a=="pop")
        pop();
        if(a=="empty")
        {
            if(empty()) cout<<"YES"<<"\n";
            else cout<<"NO"<<"\n";
        }
        if(a=="query")
        cout<<query()<<"\n";
    }
    return 0;
}
```
***  
- 此处用的是静态数组来实现模拟栈
- 所以出栈的效果和真实栈不同，它还在数组里面，假装没看到
- 其他的就没什么好说的了
